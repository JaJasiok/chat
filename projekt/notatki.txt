#include <stdio.h>
#include <unistd.h>
#include <sys/wait.h>
#include <sys/types.h>
#include <stdlib.h>
#include <string.h>
#include <signal.h>
#include <sys/types.h>   
#include <sys/ipc.h> 
#include <sys/msg.h>

struct msgbuf{
    long typ;
    char tab[1024];
    
} m1, m2;

int main(){
    int id = msgget(123465, 0644 | IPC_CREAT);
    strcpy(m1.tab, "Hello");
    m1.typ = 5;
    msgsnd(id, &m1, strlen(m1.tab)+1, 0);
    msgrcv(id, &m2, 1024, 5, 0);
    printf("%s\n", m2.tab);
    msgctl(id, IPC_RMID, NULL);
    return 0;
}
